The UC core supplies the common syntax, static protocol structure, execution
semantics, ideal-world construction, and security predicates used throughout
the project.  It is intentionally smaller than full UC: identities are natural
numbers, machines exchange at most one envelope per activation, execution is
uniform in a single security parameter, corruption is static, and the
controller uses a fixed activation budget.

The presentation below follows the dependency order of the formalization.
Each implemented object is linked to its Lean declaration, while the main text
uses mathematical notation suitable for defining and proving security of later
case studies.

\subsection{Identities, Ports, and Messages}

\begin{definition}[Machine identities]
\label{def:uc-machine-identities}
\lean{LeanCryptoProtocols.UC.MachineId, LeanCryptoProtocols.UC.env_id, LeanCryptoProtocols.UC.adv_id, LeanCryptoProtocols.UC.env_id_eq_zero, LeanCryptoProtocols.UC.adv_id_eq_one}
\leanok
The identity space is
\[
  \Id:=\Nat.
\]
The core reserves
\[
  \Env:=0,
  \qquad
  \Adv:=1
\]
for the environment and adversary.  Protocol-machine identities are required
to be distinct from both reserved identities.  An identity carries no session,
party, call-stack, or corruption information by itself.
\end{definition}

\begin{definition}[Port labels]
\label{def:uc-port-labels}
\lean{LeanCryptoProtocols.UC.PortLabel}
\leanok
The port-label space is
\[
  \Lab:=
  \{\mathsf{input},\mathsf{subout},\mathsf{backdoor}\}.
\]
An input port is used from a caller toward a subroutine.  A subroutine-output
port is used from a subroutine toward its caller or an external identity.  A
backdoor port carries adversarial control traffic.  Corruption is represented
by a message instruction, not by an additional port label.
\end{definition}

\begin{definition}[Communication ports]
\label{def:uc-comm-port}
\uses{def:uc-machine-identities,def:uc-port-labels}
\lean{LeanCryptoProtocols.UC.CommPort, LeanCryptoProtocols.UC.mk_input_port, LeanCryptoProtocols.UC.mk_subroutine_output_port, LeanCryptoProtocols.UC.mk_backdoor_port}
\leanok
A communication port is a triple
\[
  p=(p.\mathsf{own},p.\mathsf{dst},p.\mathsf{label})
  \in\Id\times\Id\times\Lab.
\]
Write \(\Port\) for the set of ports satisfying
\[
\begin{aligned}
  p.\mathsf{own}&\ne p.\mathsf{dst},\\
  p.\mathsf{label}=\mathsf{backdoor}
  &\Longleftrightarrow
  \bigl(p.\mathsf{own}=\Adv\lor p.\mathsf{dst}=\Adv\bigr).
\end{aligned}
\]
Thus every backdoor has the adversary as one endpoint, and every port incident
to the adversary is a backdoor.  The owner is the machine authorized to emit an
envelope through the port; the destination determines routing.
\end{definition}

\begin{definition}[Message instructions]
\label{def:uc-message-instruction}
\uses{def:uc-machine-identities}
\lean{LeanCryptoProtocols.UC.MessageInstruction}
\leanok
The instruction space is
\[
\begin{aligned}
  \Instr:=\;&\{\mathsf{plain}\}
  \cup\{\mathsf{corrupt}(j)\mid j\in\Id\}\\
  &\cup\{\mathsf{dummyCaller}(j)\mid j\in\Id\}
  \cup\{\mathsf{dummyDestination}(j)\mid j\in\Id\}.
\end{aligned}
\]
The instruction \(\mathsf{plain}\) denotes ordinary protocol or control data.
The instruction \(\mathsf{corrupt}(j)\) requests the corruption semantics for
identity \(j\).  The last two instructions are internal routing metadata used
by automatically generated dummy parties: one records the external caller and
the other records the external return destination.
\end{definition}

\begin{definition}[Messages]
\label{def:uc-message}
\uses{def:uc-machine-identities,def:uc-port-labels,def:uc-message-instruction}
\lean{LeanCryptoProtocols.UC.Message, LeanCryptoProtocols.UC.Message.corrupt_instruction_implies_backdoor, LeanCryptoProtocols.UC.Message.dummy_caller_instruction_implies_input, LeanCryptoProtocols.UC.Message.dummy_destination_instruction_implies_subroutine_output}
\leanok
Fix a payload space \(\Pay\).  For any set \(X\), let
\[
  \mathsf{Option}(X):=
  \{\mathsf{none}\}\cup\{\mathsf{some}(x)\mid x\in X\}.
\]
A message \(m\in\Msg(\Pay)\) consists of
\[
  m=(m.\mathsf{src},m.\mathsf{label},m.\mathsf{instr},m.\mathsf{body}),
\]
where
\[
\begin{aligned}
  m.\mathsf{src}&\in\mathsf{Option}(\Id),
  &m.\mathsf{label}&\in\Lab,\\
  m.\mathsf{instr}&\in\Instr,
  &m.\mathsf{body}&\in\Pay.
\end{aligned}
\]
The following implications are part of message well-formedness:
\[
\begin{aligned}
  m.\mathsf{instr}=\mathsf{corrupt}(j)
    &\Longrightarrow m.\mathsf{label}=\mathsf{backdoor},\\
  m.\mathsf{instr}=\mathsf{dummyCaller}(j)
    &\Longrightarrow m.\mathsf{label}=\mathsf{input},\\
  m.\mathsf{instr}=\mathsf{dummyDestination}(j)
    &\Longrightarrow m.\mathsf{label}=\mathsf{subout}.
\end{aligned}
\]
When a machine constructs an outgoing message, its source field is not yet the
authenticated network source.  The controller validates or replaces this field
before delivery, as defined in
Definition~\ref{def:uc-authenticated-message}.
\end{definition}

\begin{definition}[Envelopes]
\label{def:uc-envelope}
\uses{def:uc-comm-port,def:uc-message}
\lean{LeanCryptoProtocols.UC.Envelope}
\leanok
An envelope over \(\Pay\) is a pair
\[
  e=(e.\mathsf{port},e.\mathsf{msg})=(p,m)
  \in\Envelope(\Pay)
\]
such that
\[
  p.\mathsf{label}=m.\mathsf{label}.
\]
The controller routes using \(p.\mathsf{dst}\).  The destination machine
receives the authenticated form of \(m\), not the port object.
\end{definition}

\subsection{Machines and Protocols}

\begin{definition}[Activation results]
\label{def:uc-activation-result}
\uses{def:uc-envelope}
\lean{LeanCryptoProtocols.UC.ActivationResult}
\leanok
For a state space \(S\), define
\[
  \ActRes(S,\Pay):=
  S\times\mathsf{Option}(\Envelope(\Pay)).
\]
An activation result \((s',o)\) contains the new local state and either no
outgoing envelope or exactly one outgoing envelope.  Longer conversations are
represented by subsequent activations and retained local state.
\end{definition}

\begin{definition}[Machine programs]
\label{def:uc-machine-program}
\uses{def:uc-message,def:uc-activation-result}
\lean{LeanCryptoProtocols.UC.MachineProgram}
\leanok
A machine program with output space \(O\) consists of a local state space
\(S\) and operations
\[
\begin{array}{rcl}
  \mathsf{init} &:& \Nat\to S,\\
  \mathsf{activate} &:&
    S\to\mathsf{Option}(\Msg(\Pay))
      \to\PMF(\ActRes(S,\Pay)),\\
  \mathsf{halted} &:& S\to\Bool,\\
  \mathsf{output} &:& S\to O.
\end{array}
\]
Initialization is deterministic and indexed by the security parameter.  All
probabilistic local behavior, including sampling private protocol randomness,
occurs inside \(\mathsf{activate}\).  The optional input is
\(\mathsf{none}\) when the machine is activated without an incoming message.
There is no separate receive transition in the current core.
\end{definition}

\begin{definition}[Machines]
\label{def:uc-machine}
\uses{def:uc-machine-identities,def:uc-comm-port,def:uc-machine-program}
\lean{LeanCryptoProtocols.UC.Machine}
\leanok
A machine is a tuple
\[
  \mu=(\mu.\mathsf{id},\Gamma_\mu,P_\mu),
\]
where \(\mu.\mathsf{id}\in\Id\), \(\Gamma_\mu\subseteq\Port\) is finite,
and \(P_\mu\) is a machine program.  It satisfies
\[
\begin{aligned}
  &\forall p\in\Gamma_\mu,
    \quad p.\mathsf{own}=\mu.\mathsf{id},\\
  &\forall p,q\in\Gamma_\mu,
    \quad p.\mathsf{dst}=q.\mathsf{dst}\Longrightarrow p=q.
\end{aligned}
\]
Consequently, a machine owns every static port in its communication set and
has at most one static port to each destination.
\end{definition}

\begin{definition}[Heterogeneous machines and states]
\label{def:uc-any-machine}
\uses{def:uc-machine}
\lean{LeanCryptoProtocols.UC.AnyMachine, LeanCryptoProtocols.UC.AnyMachineState, LeanCryptoProtocols.UC.AnyMachine.id, LeanCryptoProtocols.UC.default_machine_states}
\leanok
Because protocol machines may have different output and state spaces, the
formalization uses dependent sums
\[
\begin{aligned}
  \mathsf{AnyMachine}(\Pay)
    &:=\sum_{O:\mathsf{Type}}\mathsf{Machine}(\Pay,O),\\
  \mathsf{AnyState}(\Pay)
    &:=\sum_{\mu:\mathsf{AnyMachine}(\Pay)}S_\mu.
\end{aligned}
\]
For a machine list \(L=[\mu_1,\ldots,\mu_k]\), deterministic initialization is
\[
  \mathsf{InitStates}(L,n)
  :=[(\mu_i,\mu_i.\mathsf{init}(n))]_{i=1}^{k}.
\]
The dependent packaging is an implementation device; each packaged state still
belongs to the program of the packaged machine.
\end{definition}

\begin{definition}[Port-induced call relations]
\label{def:uc-port-relations}
\uses{def:uc-machine,def:uc-port-labels}
\lean{LeanCryptoProtocols.UC.can_send_input_to, LeanCryptoProtocols.UC.can_send_subroutine_output_to, LeanCryptoProtocols.UC.can_send_backdoor_to, LeanCryptoProtocols.UC.is_caller_of_id, LeanCryptoProtocols.UC.is_subroutine_of_id}
\leanok
For a machine \(\mu\) and identity \(j\), write
\[
\begin{aligned}
  \mu\leadsto_{\mathsf{in}}j
  &\Longleftrightarrow
    \exists p\in\Gamma_\mu,\;
      p.\mathsf{dst}=j\land p.\mathsf{label}=\mathsf{input},\\
  \mu\leadsto_{\mathsf{sub}}j
  &\Longleftrightarrow
    \exists p\in\Gamma_\mu,\;
      p.\mathsf{dst}=j\land p.\mathsf{label}=\mathsf{subout},\\
  \mu\leadsto_{\mathsf{bd}}j
  &\Longleftrightarrow
    \exists p\in\Gamma_\mu,\;
      p.\mathsf{dst}=j\land p.\mathsf{label}=\mathsf{backdoor}.
\end{aligned}
\]
The first relation makes \(\mu\) a caller of \(j\); the second makes \(\mu\)
a subroutine of \(j\).
\end{definition}

\begin{definition}[Protocols]
\label{def:uc-protocol}
\uses{def:uc-machine-identities,def:uc-any-machine,def:uc-port-relations}
\lean{LeanCryptoProtocols.UC.Protocol, LeanCryptoProtocols.UC.machine_ids, LeanCryptoProtocols.UC.Protocol.has_machine_id, LeanCryptoProtocols.UC.Protocol.machine_by_id?}
\leanok
A protocol \(\Prot\) is a finite machine list
\[
  \Prot.\mathsf{machines}=[\mu_1,\ldots,\mu_k]
\]
with ordered identity list and underlying finite identity set
\[
\begin{aligned}
  \IdList(\Prot)&:=[\mu_1.\mathsf{id},\ldots,\mu_k.\mathsf{id}],\\
  \Ids(\Prot)&:=\{\mu_1.\mathsf{id},\ldots,\mu_k.\mathsf{id}\}.
\end{aligned}
\]
The record proves the following static obligations:
\[
\begin{aligned}
  &\IdList(\Prot)\text{ has no duplicate},\\
  &\Env\notin\Ids(\Prot),
   \qquad \Adv\notin\Ids(\Prot),\\
  &\forall\mu\in\Prot,\;\forall j\in\Id,\quad
    \mu\leadsto_{\mathsf{in}}j
    \Longrightarrow
    \exists\nu\in\Prot,\;
      \nu.\mathsf{id}=j
      \land\nu\leadsto_{\mathsf{sub}}\mu.\mathsf{id},\\
  &\forall\mu\in\Prot,\;\forall j\in\Id,\quad
    \mu\leadsto_{\mathsf{sub}}j\land j\in\Ids(\Prot)
    \Longrightarrow
    \exists\nu\in\Prot,\;
      \nu.\mathsf{id}=j
      \land\nu\leadsto_{\mathsf{in}}\mu.\mathsf{id},\\
  &\forall\mu\in\Prot,\;\forall p\in\Gamma_\mu,
    \quad p.\mathsf{dst}\ne\Env.
\end{aligned}
\]
The protocol contains no protocol-level random initial-state sampler.  At the
start of an execution every listed machine is initialized by its own
deterministic \(\mathsf{init}(n)\).
\end{definition}

\begin{definition}[Public-setup convention]
\label{rem:uc-public-setup}
\uses{def:uc-machine-program,def:uc-protocol}
The core has no first-class random public-setup field.  Fixed public data, such
as a group family, generator, or algebraic operations, may parameterize the
Lean value used to construct a protocol and its machine programs.  Per-session
private randomness is sampled by machine activations.  A randomized common
setup, such as a CRS sampled once and shared by multiple machines, must be
modeled explicitly by a setup machine or ideal functionality if it is to be
part of the execution semantics.
\end{definition}

\begin{definition}[Main, internal, and external identities]
\label{def:uc-main-internal-external}
\uses{def:uc-protocol,def:uc-port-relations}
\lean{LeanCryptoProtocols.UC.Protocol.is_external_identity_of, LeanCryptoProtocols.UC.Protocol.is_main_machine, LeanCryptoProtocols.UC.Protocol.is_internal_machine, LeanCryptoProtocols.UC.Protocol.external_identities_of, LeanCryptoProtocols.UC.Protocol.main_machine_ids, LeanCryptoProtocols.UC.Protocol.internal_machine_ids, LeanCryptoProtocols.UC.Protocol.machine_is_subroutine_of, LeanCryptoProtocols.UC.Protocol.machine_is_caller_of, LeanCryptoProtocols.UC.Protocol.machine_is_subsidiary_of}
\leanok
An identity \(x\) is external to a protocol machine \(\mu\) when
\[
  \mu\in\Prot,
  \qquad
  \mu\leadsto_{\mathsf{sub}}x,
  \qquad
  x\notin\Ids(\Prot).
\]
Write \(x\in\Ext_\Prot(\mu)\) for this relation.  For a machine identity
\(j\), define
\[
  \Ext_\Prot(j):=
  \{x\in\Id\mid
    \exists\mu\in\Prot,\;
      \mu.\mathsf{id}=j\land x\in\Ext_\Prot(\mu)\}.
\]
Identity uniqueness ensures that this set belongs to at most one listed
machine.  A protocol-machine identity \(j\) is main when its external-identity
set is nonempty:
\[
  j\in\Main(\Prot)
  \Longleftrightarrow
  j\in\Ids(\Prot)\land\Ext_\Prot(j)\ne\varnothing.
\]
An internal machine is a listed machine that is not main.  Direct
subroutine and caller relations are induced by ports; the subsidiary relation
is the transitive closure of the direct subroutine relation.
\end{definition}

\subsection{Execution Setups and Adversarial Control}

\begin{definition}[Environment]
\label{def:uc-environment}
\uses{def:uc-machine-identities,def:uc-machine}
\lean{LeanCryptoProtocols.UC.Environment}
\leanok
An environment \(E\) is a machine with identity \(\Env\), output space
\(\Bool\), and a static communication set consisting of exactly one backdoor
port to \(\Adv\).  Input ports from \(E\) to main protocol machines are added
at execution time and are not part of this static singleton.
\end{definition}

\begin{definition}[Adversary]
\label{def:uc-adversary}
\uses{def:uc-machine-identities,def:uc-machine}
\lean{LeanCryptoProtocols.UC.Adversary}
\leanok
An adversary \(A\) is a machine with identity \(\Adv\), unit output, and a
static communication set consisting of exactly one backdoor port to \(\Env\).
Backdoor ports from \(A\) to protocol machines are supplied by the runtime
overlay.
\end{definition}

\begin{definition}[Simulator]
\label{def:uc-simulator}
\uses{def:uc-adversary}
\lean{LeanCryptoProtocols.UC.Simulator}
\leanok
A simulator has the same machine type as an adversary:
\[
  \mathsf{Simulator}(\Pay):=\mathsf{Adversary}(\Pay).
\]
It occupies the adversarial slot in an ideal execution.  This abbreviation does
not prevent a simulator witness from containing or invoking a given real-world
adversary: in the UC quantifier order the simulator is selected after that
adversary.
\end{definition}

\begin{definition}[Execution setup and static corruption]
\label{def:uc-execution-setup}
\uses{def:uc-protocol,def:uc-environment,def:uc-adversary}
\lean{LeanCryptoProtocols.UC.ExecutionSetup}
\leanok
For a protocol \(\Prot\), adversary \(A\), and environment \(E\), an execution
setup
\[
  \sigma\in\Setup(\Prot,A,E)
\]
stores a finite static corruption pattern \(\Corr_\sigma\subseteq\Id\) and a
proof that
\[
  \Corr_\sigma\subseteq\Ids(\Prot).
\]
There are no other setup fields.  In particular, the adversary object does not
store its own corruption set.  The current constraint permits any listed
protocol machine, including an internal machine, to occur in \(\Corr_\sigma\);
it is not restricted to main-machine identities.
\end{definition}

\begin{definition}[Runtime communication sets]
\label{def:uc-runtime-overlay}
\uses{def:uc-comm-port,def:uc-main-internal-external,def:uc-execution-setup}
\lean{LeanCryptoProtocols.UC.ExecutionSetup.protocol_comm_set, LeanCryptoProtocols.UC.ExecutionSetup.backdoor_port_to_adversary, LeanCryptoProtocols.UC.ExecutionSetup.backdoor_port_from_adversary, LeanCryptoProtocols.UC.ExecutionSetup.env_input_port_to_main, LeanCryptoProtocols.UC.ExecutionSetup.runtime_communication_set}
\leanok
For a setup \(\sigma\), the controller computes the ports available to active
sender \(i\) as a finite set \(\Runtime_\sigma(i)\).  Let \(\Gamma_i\) be the
static communication set of machine \(i\), when such a machine exists.  Then
\[
\begin{aligned}
  \Runtime_\sigma(\Env)
  ={}&\Gamma_E\cup
    \{(\Env,j,\mathsf{input})\mid j\in\Main(\Prot)\},\\
  \Runtime_\sigma(\Adv)
  ={}&\Gamma_A\cup
    \{(\Adv,j,\mathsf{backdoor})\mid j\in\Ids(\Prot)\},\\
  \Runtime_\sigma(i)
  ={}&\Gamma_i\cup\{(i,\Adv,\mathsf{backdoor})\},
    \quad i\in\Ids(\Prot),\\
  \Runtime_\sigma(i)
  ={}&\varnothing,
    \quad i\notin\{\Env,\Adv\}\cup\Ids(\Prot).
\end{aligned}
\]
The overlay grants ordinary control communication with every protocol machine.
It does not by itself implement corruption or force a target program to react.
\end{definition}

\begin{definition}[System and external routing targets]
\label{def:uc-route-targets}
\uses{def:uc-main-internal-external,def:uc-execution-setup}
\lean{LeanCryptoProtocols.UC.ExecutionSetup.routes_to_system_machine, LeanCryptoProtocols.UC.ExecutionSetup.routes_to_external_identity}
\leanok
An identity \(j\) is a system destination exactly when
\[
  j=\Env\lor j=\Adv\lor j\in\Ids(\Prot).
\]
It is an external destination when it is external to at least one listed
protocol machine.  System destinations denote controller-managed machine
states.  External destinations are logical caller identities; output addressed
to them is delivered to the environment.
\end{definition}

\begin{definition}[Environment source validity]
\label{def:uc-environment-source}
\uses{def:uc-envelope,def:uc-main-internal-external,def:uc-execution-setup}
\lean{LeanCryptoProtocols.UC.ExecutionSetup.environment_source_valid}
\leanok
For an envelope \(e=(p,m)\) emitted by \(E\), define
\[
  \mathsf{EnvSrcValid}_\sigma(e)
  \Longleftrightarrow
  \begin{cases}
    \top,
      &p.\mathsf{dst}=\Adv,\\
    \exists x\in\Ext_\Prot(p.\mathsf{dst}),\;
      m.\mathsf{src}=\mathsf{some}(x),
      &p.\mathsf{dst}\ne\Adv.
  \end{cases}
\]
The second branch lets the environment choose which valid external identity is
calling a main machine.  This logical caller identity is the only source value
that the controller preserves rather than replacing by the active machine
identity.
\end{definition}

\begin{definition}[Corruption-instruction validity]
\label{def:uc-corruption-validity}
\uses{def:uc-message-instruction,def:uc-execution-setup}
\lean{LeanCryptoProtocols.UC.ExecutionSetup.corruption_instruction_valid, LeanCryptoProtocols.UC.ExecutionSetup.corrupt_instruction_sender_is_adversary}
\leanok
For sender \(i\) and envelope \(e=(p,m)\), let
\[
  \CorrValid_\sigma(i,e):=\top
\]
when \(m.\mathsf{instr}\) is \(\mathsf{plain}\),
\(\mathsf{dummyCaller}(x)\), or \(\mathsf{dummyDestination}(x)\), for any
\(x\in\Id\).  The only nontrivial branch is
\[
\begin{aligned}
  &m.\mathsf{instr}=\mathsf{corrupt}(j)\\
  &\qquad\Longrightarrow\quad
  \bigl(\CorrValid_\sigma(i,e)
  \Longleftrightarrow
  i=\Adv\land m.\mathsf{label}=\mathsf{backdoor}
  \land j\in\Corr_\sigma\bigr).
\end{aligned}
\]
Thus only the adversary or simulator may issue a valid corruption instruction.
The instruction target \(j\) need not equal \(p.\mathsf{dst}\).  This separation
is necessary in the ideal world, where the simulator may address the ideal
functionality while naming a corrupted dummy-party identity in the instruction.
The receiving program determines the resulting leakage or command semantics.
\end{definition}

\begin{definition}[Outgoing-envelope validity]
\label{def:uc-outgoing-validity}
\uses{def:uc-runtime-overlay,def:uc-environment-source,def:uc-corruption-validity}
\lean{LeanCryptoProtocols.UC.ExecutionSetup.outgoing_message_valid}
\leanok
An envelope \(e=(p,m)\) emitted by active sender \(i\) is accepted exactly when
\[
\begin{aligned}
  \OutValid_\sigma(i,e)
  \Longleftrightarrow{}&
  p.\mathsf{own}=i
  \land p\in\Runtime_\sigma(i)\\
  &\land\CorrValid_\sigma(i,e)
  \land
  \bigl(i=\Env\Longrightarrow
    \mathsf{EnvSrcValid}_\sigma(e)\bigr).
\end{aligned}
\]
The first two conjuncts authenticate the actual emitting machine and its route.
An invalid envelope is dropped and is never delivered.
\end{definition}

\begin{definition}[Authenticated delivered messages]
\label{def:uc-authenticated-message}
\uses{def:uc-outgoing-validity,def:uc-message}
\lean{LeanCryptoProtocols.UC.ExecutionSetup.authenticated_message, LeanCryptoProtocols.UC.ExecutionSetup.authenticated_message_of_sender_ne_environment, LeanCryptoProtocols.UC.ExecutionSetup.authenticated_message_environment_to_adversary, LeanCryptoProtocols.UC.ExecutionSetup.authenticated_message_environment_to_non_adversary}
\leanok
Let \(e=(p,m)\) be a valid envelope from active sender \(i\).  The message
delivered by the controller is
\[
  \Auth_\sigma(i,e):=
  \begin{cases}
    m[m.\mathsf{src}:=\mathsf{some}(\Env)],
      &i=\Env\land p.\mathsf{dst}=\Adv,\\
    m,
      &i=\Env\land p.\mathsf{dst}\ne\Adv,\\
    m[m.\mathsf{src}:=\mathsf{some}(i)],
      &i\ne\Env.
  \end{cases}
\]
Consequently, a receiver cannot observe a forged P2P machine source.  The only
preserved non-machine source is the environment-selected external identity,
which has already passed
Definition~\ref{def:uc-environment-source}.
\end{definition}

\subsection{Controller Execution}

\begin{definition}[Protocol-machine runtime states]
\label{def:uc-protocol-runtime-state}
\uses{def:uc-any-machine,def:uc-activation-result}
\lean{LeanCryptoProtocols.UC.ProtocolMachineState, LeanCryptoProtocols.UC.ProtocolMachineState.of_any_machine_state, LeanCryptoProtocols.UC.ProtocolMachineState.id, LeanCryptoProtocols.UC.ProtocolMachineState.activate}
\leanok
A protocol-machine runtime state packages a machine \(\mu\) with a value
\(s\in S_\mu\).  Activating it on optional input \(m?\) samples
\[
  (s',o)\leftarrow
  \mu.\mathsf{activate}(s,m?),
\]
then returns the same machine packaged with \(s'\), together with \(o\).  The
machine program and identity are unchanged by activation.
\end{definition}

\begin{definition}[Controller states and initialization]
\label{def:uc-controller-state}
\uses{def:uc-execution-setup,def:uc-protocol-runtime-state}
\lean{LeanCryptoProtocols.UC.ControllerState, LeanCryptoProtocols.UC.Controller.initial_state, LeanCryptoProtocols.UC.Controller.find_protocol_state?, LeanCryptoProtocols.UC.Controller.replace_protocol_state}
\leanok
A controller state for setup \(\sigma\) is
\[
  s=(s_E,s_A,\vec{s}_{\Prot},i),
\]
where \(s_E\) and \(s_A\) are the environment and adversary local states,
\(\vec{s}_{\Prot}\) contains one initialized state for each listed protocol
machine, and \(i\in\Id\) is the active identity.  At security parameter \(n\),
the initial state is
\[
\begin{aligned}
  s_E&:=E.\mathsf{init}(n),
  &s_A&:=A.\mathsf{init}(n),\\
  \vec{s}_{\Prot}&:=
    [\mu.\mathsf{init}(n)\mid\mu\in\Prot],
  &i&:=\Env.
\end{aligned}
\]
Lookup and replacement of protocol states use machine identities.  Identity
uniqueness makes the intended replacement target unambiguous.
The raw controller-state type does not itself carry a proof that
\(\vec{s}_\Prot\) contains exactly one state for every machine in \(\Prot\).
The designated initializer constructs precisely that list, and executions
produced by \(\mathsf{exec}\) start from it, but this correspondence is not
currently exposed as a separately proved controller invariant.
\end{definition}

\begin{definition}[Activation of the active component]
\label{def:uc-component-activation}
\uses{def:uc-controller-state,def:uc-machine-program}
\lean{LeanCryptoProtocols.UC.Controller.activate_environment, LeanCryptoProtocols.UC.Controller.activate_adversary, LeanCryptoProtocols.UC.Controller.activate_protocol_machine, LeanCryptoProtocols.UC.Controller.activate_machine}
\leanok
Given controller state \(s\), active identity \(i\), and optional incoming
message \(m?\), the controller invokes exactly one program:
\[
\begin{array}{c|c}
  i & \text{program activated}\\ \hline
  \Env & E,\\
  \Adv & A,\\
  i\in\Ids(\Prot) & \text{the unique protocol machine with identity }i.
\end{array}
\]
The returned local state replaces only the active component's state.  If a
nonreserved identity has no runtime protocol state, activation emits no envelope
and returns control to \(\Env\).
\end{definition}

\begin{definition}[Bounded activation loop]
\label{def:uc-run-activation}
\uses{def:uc-component-activation,def:uc-outgoing-validity,def:uc-authenticated-message,def:uc-route-targets}
\lean{LeanCryptoProtocols.UC.Controller.environment_halted, LeanCryptoProtocols.UC.Controller.environment_output, LeanCryptoProtocols.UC.Controller.run_activation}
\leanok
Let \(\RunAct_\sigma(k,s,m?)\) denote the controller distribution with \(k\)
remaining activations.  If \(k=0\), or if the environment has halted, it returns
\(s\).  Otherwise it performs the following transition:
\begin{enumerate}
\item Let \(i=s.\mathsf{active}\).  Activate \(i\) on \(m?\), sample a new
      local state and optional envelope, and install the new local state.
\item If no envelope is emitted, set the active identity to \(\Env\) and recurse
      with \(k-1\) and input \(\mathsf{none}\).
\item If envelope \(e\) is emitted but \(\OutValid_\sigma(i,e)\) is false,
      drop \(e\), set the active identity to \(\Env\), and recurse with no input.
\item Otherwise form \(m':=\Auth_\sigma(i,e)\) and route as follows:
\end{enumerate}
Put \(d:=e.\mathsf{port}.\mathsf{dst}\).  The next active identity and input are
the pair
\[
  \mathsf{Route}_\sigma(i,e,m'):=
  \begin{cases}
    (\Env,\mathsf{some}(m')), & d=\Env,\\
    (\Adv,\mathsf{some}(m')), & d=\Adv,\\
    (d,\mathsf{some}(m')), & d\in\Ids(\Prot),\\
    (\Env,\mathsf{some}(m')),
      &
      \begin{aligned}[t]
        &i\ne\Env,\ d\text{ is external to }\Prot,\\
        &e.\mathsf{msg}.\mathsf{label}=\mathsf{subout},
      \end{aligned}\\
    (\Env,\mathsf{none}), & \text{otherwise}.
  \end{cases}
\]
The controller installs this pair and recurses with \(k-1\).  Delivery is not a
separate receive operation: the authenticated message becomes the optional
input of the receiving machine's next activation.  Each such activation consumes
one unit of fuel.
\end{definition}

\begin{definition}[Execution ensemble]
\label{def:uc-execution-ensemble}
\uses{def:uc-controller-state,def:uc-run-activation}
\lean{LeanCryptoProtocols.UC.Controller.ExitStatus, LeanCryptoProtocols.UC.Controller.run_steps, LeanCryptoProtocols.UC.Controller.run_steps_with_status, LeanCryptoProtocols.UC.Controller.exec, LeanCryptoProtocols.UC.Controller.exec_with_status, LeanCryptoProtocols.UC.Controller.budget_exceeded_warning, LeanCryptoProtocols.UC.Controller.warn_if_budget_exceeded}
\leanok
Let \(T\) be the project-wide maximum activation count.  Starting from the
state of Definition~\ref{def:uc-controller-state}, execution runs
\(\RunAct_\sigma(T,\cdot,\mathsf{none})\) and extracts the environment's current
Boolean output.  Thus every setup induces an ensemble
\[
  \Run_\sigma:\Nat\to\PMF(\Bool).
\]
The security parameter \(n\) controls machine initialization; it is not the
fuel.  The status variant additionally reports whether the environment halted
or the fixed budget was exhausted.  The ordinary security ensemble returns the
environment output in either case.
\end{definition}

\begin{definition}[Execution-model boundary]
\label{rem:uc-execution-boundaries}
\uses{def:uc-execution-ensemble}
The controller budget is a fixed project constant rather than an explicit
polynomial in \(n\).  The current execution model also has no auxiliary-input
parameter.  These are deliberate boundaries of the implemented restricted
model, not consequences of the general UC framework.
\end{definition}

\subsection{Ideal Functionalities and Generated Ideal Protocols}

\begin{definition}[Ideal functionalities]
\label{def:uc-ideal-functionality}
\uses{def:uc-machine,def:uc-main-internal-external}
\lean{LeanCryptoProtocols.UC.IdealFunctionality}
\leanok
An ideal functionality \(F\) records
\[
  P_F=[j_1,\ldots,j_t],
  \qquad X_F(j)\subseteq\Id,
  \qquad f\in\Id,
  \qquad \mu_F,
\]
where \(P_F\) is the party-identity list, \(X_F(j)\) is the finite set of
external identities associated with party \(j\), \(f\) is the functionality
identity, and \(\mu_F\) is a unit-output machine with identity \(f\).  The
record proves:
\[
\begin{aligned}
  &P_F\text{ has no duplicate},
  &&\Env,\Adv\notin P_F,\\
  &f\notin\{\Env,\Adv\},
  &&f\notin P_F,\\
  &X_F(j)\ne\varnothing
    &&\text{for every }j\in P_F,\\
  &x\in X_F(j)
    &&\Longrightarrow
      x\notin P_F\land x\notin\{f,\Env,\Adv\},\\
  &\forall j\in P_F,\;\exists p\in\Gamma_{\mu_F},
    &&p.\mathsf{dst}=j\land p.\mathsf{label}=\mathsf{subout},\\
  &\forall p\in\Gamma_{\mu_F},
    &&\bigl(p.\mathsf{dst}\in P_F
      \land p.\mathsf{label}=\mathsf{subout}\bigr)\\
  &&&\quad\lor
      \bigl(p.\mathsf{dst}=\Adv
      \land p.\mathsf{label}=\mathsf{backdoor}\bigr).
\end{aligned}
\]
The functionality program itself defines all ideal behavior, including normal
outputs, scheduling leakage, corruption leakage, and accepted simulator
commands.
\end{definition}

\begin{definition}[Generated dummy parties]
\label{def:uc-dummy-party}
\uses{def:uc-ideal-functionality,def:uc-authenticated-message}
\lean{LeanCryptoProtocols.UC.DummyParty, LeanCryptoProtocols.UC.DummyParty.to_any_machine, LeanCryptoProtocols.UC.DummyPartyImpl.input_port, LeanCryptoProtocols.UC.DummyPartyImpl.output_port, LeanCryptoProtocols.UC.DummyPartyImpl.communication_set, LeanCryptoProtocols.UC.DummyPartyImpl.program, LeanCryptoProtocols.UC.mk_dummy_party, LeanCryptoProtocols.UC.mk_dummy_parties, LeanCryptoProtocols.UC.mk_dummy_parties_party_ids}
\leanok
For each \(j\in P_F\), the core generates a unit-state machine
\[
  D_j:=\Dummy(F,j).
\]
Its static communication set consists of one input port from \(j\) to \(f\)
and one subroutine-output port from \(j\) to every \(x\in X_F(j)\).  It has no
static backdoor port.  Its activation behavior is exactly:
\begin{enumerate}
\item On an input message with authenticated source \(x\), instruction
      \(\mathsf{plain}\), and payload \(u\), emit the same payload to \(F\) with
      instruction \(\mathsf{dummyCaller}(x)\).
\item On a subroutine-output message carrying
      \(\mathsf{dummyDestination}(x)\), emit the same payload through the output
      port to \(x\), if \(x\in X_F(j)\), and replace the instruction by
      \(\mathsf{plain}\).
\item On no input, a backdoor message, or any other instruction-label shape,
      emit no envelope.
\end{enumerate}
When the first outgoing message reaches \(F\), the controller authenticates its
source as \(j\); the external caller remains separately recorded in
\(\mathsf{dummyCaller}(x)\).  When the second outgoing message reaches the
environment, its authenticated source is again \(j\).
\end{definition}

\begin{definition}[Ideal protocol generated from a functionality]
\label{def:uc-ideal-protocol}
\uses{def:uc-protocol,def:uc-dummy-party}
\lean{LeanCryptoProtocols.UC.IdealProtocol, LeanCryptoProtocols.UC.mk_ideal_protocol, LeanCryptoProtocols.UC.mk_ideal_protocol_machine_ids, LeanCryptoProtocols.UC.mk_ideal_protocol_has_machine_id_iff, LeanCryptoProtocols.UC.mk_ideal_protocol_is_main_machine_iff, LeanCryptoProtocols.UC.mk_ideal_protocol_is_internal_machine_iff}
\leanok
The generated ideal protocol is
\[
  \Phi_F:=\IdealProtocol(F)
\]
with machine list
\[
  [D_j\mid j\in P_F]\mathbin{++}[\mu_F].
\]
Its identities are exactly \(P_F\mathbin{++}[f]\).  The constructor proves
\[
\begin{aligned}
  j\in\Main(\Phi_F)&\Longleftrightarrow j\in P_F,\\
  j\text{ is internal in }\Phi_F&\Longleftrightarrow j=f.
\end{aligned}
\]
Thus the environment-facing parties are precisely the generated dummies, and
the only internal machine is the ideal functionality.
\end{definition}

\begin{definition}[Simulator access in the generated ideal world]
\label{rem:uc-ideal-simulator-access}
\uses{def:uc-runtime-overlay,def:uc-corruption-validity,def:uc-dummy-party,def:uc-ideal-protocol}
The runtime overlay creates backdoor routes between the simulator and every
machine in \(\Phi_F\), including dummy parties.  This routing fact is not direct
control: generated dummy programs ignore every backdoor input.  Effective
simulator interaction therefore occurs through behavior explicitly implemented
by \(F\).  A simulator may, for example, send
\(\mathsf{corrupt}(j)\) to \(F\) while naming a dummy identity
\(j\in\Corr_\sigma\); the functionality decides what this reveals or permits.
Because the current setup constraint ranges over all ideal-protocol identities,
it also permits the functionality identity itself to occur in the corruption
set, although party-corruption theorems will normally choose only dummy-party
identities.
\end{definition}

\subsection{Indistinguishability}

\begin{definition}[Ensembles]
\label{def:uc-ensemble}
\lean{LeanCryptoProtocols.UC.Ensemble}
\leanok
An ensemble over \(\alpha\) is a family
\[
  X:\Nat\to\PMF(\alpha).
\]
The index is the uniform security parameter.
\end{definition}

\begin{definition}[Abstract polynomial time]
\label{def:uc-ppt}
\lean{LeanCryptoProtocols.UC.PPT, LeanCryptoProtocols.UC.ppt_congr, LeanCryptoProtocols.UC.ppt_comp, LeanCryptoProtocols.UC.ppt_const, LeanCryptoProtocols.UC.ppt_pure, LeanCryptoProtocols.UC.ppt_bind}
\leanok
The predicate \(\PPTM(x)\) is the project's abstract polynomial-time
interface.  It is used for adversaries, simulators, environments,
distinguishers, reductions, and samplers.  The current axioms provide closure
under equality, deterministic composition, constants, pure probabilistic
return, and probabilistic bind.  They do not define an operational cost model.
\end{definition}

\begin{definition}[Negligible functions]
\label{def:uc-negligible}
\lean{LeanCryptoProtocols.UC.Negligible}
\leanok
A function \(\varepsilon:\Nat\to\mathbb{R}\) is negligible when it has
superpolynomial decay at infinity in the mathlib sense.  Write
\(\Negl(\varepsilon)\) for this property.
\end{definition}

\begin{definition}[Security levels]
\label{def:uc-security-levels}
\lean{LeanCryptoProtocols.UC.SecurityLevel}
\leanok
The supported levels are
\[
  \ell\in
  \{\mathsf{perfect},\mathsf{statistical},\mathsf{computational}\}.
\]
\end{definition}

\begin{definition}[Probability mass and total variation]
\label{def:uc-probability-distance}
\uses{def:uc-ensemble}
\lean{LeanCryptoProtocols.UC.probTrue, LeanCryptoProtocols.UC.pmfMass, LeanCryptoProtocols.UC.TVDist, LeanCryptoProtocols.UC.StatisticalIndistBound}
\leanok
For \(B\in\PMF(\Bool)\), write \(\Pr[B=1]\) for its mass at
\(\mathsf{true}\).  For finite \(\alpha\) and distributions \(P,Q\) over
\(\alpha\), define
\[
  \TV(P,Q):=
  \frac12\sum_{a\in\alpha}|P(a)-Q(a)|.
\]
An explicit statistical bound for ensembles \(X,Y\) is a function
\(\varepsilon\) satisfying
\[
  \forall n,\qquad
  \TV(X(n),Y(n))\le\varepsilon(n).
\]
\end{definition}

\begin{definition}[Perfect indistinguishability]
\label{def:uc-perfect-indist}
\uses{def:uc-ensemble}
\lean{LeanCryptoProtocols.UC.PerfectIndist}
\leanok
Perfect indistinguishability is pointwise equality:
\[
  X\equiv Y
  \Longleftrightarrow
  \forall n,\;X(n)=Y(n).
\]
\end{definition}

\begin{definition}[Statistical indistinguishability]
\label{def:uc-statistical-indist}
\uses{def:uc-ensemble,def:uc-negligible,def:uc-probability-distance}
\lean{LeanCryptoProtocols.UC.StatisticalIndist}
\leanok
Statistical indistinguishability is
\[
  X\approx_sY
  \Longleftrightarrow
  \exists\varepsilon,\;
  \Negl(\varepsilon)\land
  \forall n,\;
    \TV(X(n),Y(n))\le\varepsilon(n).
\]
\end{definition}

\begin{definition}[Uniform distinguishers]
\label{def:uc-distinguisher}
\uses{def:uc-ensemble,def:uc-probability-distance}
\lean{LeanCryptoProtocols.UC.Distinguisher, LeanCryptoProtocols.UC.DistOutput, LeanCryptoProtocols.UC.DistAdvantage}
\leanok
A uniform distinguisher for samples in \(\alpha\) is an algorithm
\[
  D:\alpha\to\Nat\to\PMF(\Bool).
\]
For an ensemble \(X\), define the induced output distribution and the
distinguishing advantage by
\[
\begin{aligned}
  \mathsf{Out}_D(X,n)
    &:=X(n).\mathsf{bind}\bigl(x\mapsto D(x,n)\bigr),\\
  \mathsf{Adv}_D(X,Y,n)
    &:=\left|
      \Pr[\mathsf{Out}_D(X,n)=1]
      -\Pr[\mathsf{Out}_D(Y,n)=1]
    \right|.
\end{aligned}
\]
\end{definition}

\begin{definition}[Computational indistinguishability]
\label{def:uc-computational-indist}
\uses{def:uc-distinguisher,def:uc-ppt,def:uc-negligible}
\lean{LeanCryptoProtocols.UC.ComputationalIndist}
\leanok
Computational indistinguishability means that every PPT distinguisher has
negligible advantage:
\[
\begin{aligned}
  X\approx_cY
  \Longleftrightarrow
  \forall D,\;&\PPTM(D)\Longrightarrow
  \exists\varepsilon,\;\Negl(\varepsilon)\land{}\\
  &\forall n,\quad
  \mathsf{Adv}_D(X,Y,n)\le\varepsilon(n).
\end{aligned}
\]
\end{definition}

\begin{definition}[Unified indistinguishability predicate]
\label{def:uc-indistinguishability}
\uses{def:uc-security-levels,def:uc-perfect-indist,def:uc-statistical-indist,def:uc-computational-indist}
\lean{LeanCryptoProtocols.UC.Indistinguishable}
\leanok
The predicate \(\mathsf{Indist}_\ell(X,Y)\) selects perfect, statistical, or
computational indistinguishability according to \(\ell\).
\end{definition}

\begin{definition}[Controller-induced distinguisher closure]
\label{def:uc-controller-ppt}
\uses{def:uc-ppt,def:uc-computational-indist,def:uc-execution-ensemble}
\lean{LeanCryptoProtocols.UC.Controller.ppt_controller_distinguisher}
\leanok
For a challenge sample \(x\), suppose a reduction constructs a protocol
\(\Prot_x\) and setup \(\sigma_x\), then returns the controller's environment
bit.  The current complexity interface postulates
\[
  \PPTM(A)\land\PPTM(E)
  \Longrightarrow
  \PPTM\bigl(x\mapsto\Run_{\sigma_x}\bigr).
\]
This axiom is a complexity-closure statement only.  It does not prove any trace
equivalence, probability bound, or cryptographic assumption.
\end{definition}

\subsection{UC Security Definitions}

\begin{definition}[Execution difference]
\label{def:uc-exec-diff}
\uses{def:uc-execution-ensemble,def:uc-probability-distance}
\lean{LeanCryptoProtocols.UC.exec_diff}
\leanok
For real and ideal setups using the same environment, define
\[
  \Delta(\sigma_r,\sigma_i,n):=
  \left|
    \Pr[\Run_{\sigma_r}(n)=1]
    -\Pr[\Run_{\sigma_i}(n)=1]
  \right|.
\]
This is the bit-output advantage already chosen by the environment; no
additional distinguisher is applied in the UC definition.
\end{definition}

\begin{definition}[UC emulation at a static corruption pattern]
\label{def:uc-emulation}
\uses{def:uc-simulator,def:uc-execution-setup,def:uc-security-levels,def:uc-negligible,def:uc-exec-diff}
\lean{LeanCryptoProtocols.UC.UCEmulatesAt, LeanCryptoProtocols.UC.UCEmulatesPerfect, LeanCryptoProtocols.UC.UCEmulatesPerfectWithCorruption, LeanCryptoProtocols.UC.UCEmulatesStatistical, LeanCryptoProtocols.UC.UCEmulatesStatisticalWithCorruption, LeanCryptoProtocols.UC.UCEmulatesComputational, LeanCryptoProtocols.UC.UCEmulatesComputationalWithCorruption}
\leanok
Fix protocols \(\Prot,\Phi\) and a finite corruption pattern \(\Corr\).  At
the perfect level, \(\UCEmul_{\mathsf{perf}}(\Corr,\Prot,\Phi)\) means
\[
\begin{aligned}
  \forall A\;\exists S\;\forall E\;
  \forall\sigma_r&\in\Setup(\Prot,A,E)\;
  \forall\sigma_i\in\Setup(\Phi,S,E),\\
  \Corr_{\sigma_r}=\Corr\land\Corr_{\sigma_i}=\Corr
  &\Longrightarrow
  \forall n,\quad
  \Run_{\sigma_r}(n)=\Run_{\sigma_i}(n).
\end{aligned}
\]
At the statistical level, the same quantifier prefix is followed by
\[
  \exists\varepsilon,\quad
  \Negl(\varepsilon)\land
  \forall n,\quad
  \Delta(\sigma_r,\sigma_i,n)\le\varepsilon(n).
\]
At the computational level, the exact quantifier order and complexity
conditions are
\[
\begin{aligned}
  \forall A,\;&\PPTM(A)\Longrightarrow
  \exists S,\;\PPTM(S)\land{}\\
  &\forall E,\;\PPTM(E)\Longrightarrow
  \forall\sigma_r\in\Setup(\Prot,A,E),\\
  &\forall\sigma_i\in\Setup(\Phi,S,E),\\
  &\Corr_{\sigma_r}=\Corr\land\Corr_{\sigma_i}=\Corr
  \Longrightarrow{}\\
  &\exists\varepsilon,\quad
  \Negl(\varepsilon)\land
  \forall n,\quad
  \Delta(\sigma_r,\sigma_i,n)\le\varepsilon(n).
\end{aligned}
\]
The same environment object appears in both setups.  The simulator depends on
the adversary but is independent of the subsequently quantified environment;
its machine program may store and black-box invoke that adversary.  The standard
no-corruption aliases instantiate \(\Corr=\varnothing\).
\end{definition}

\begin{definition}[Legality of the fixed corruption pattern]
\label{rem:uc-corruption-pattern-legality}
\uses{def:uc-execution-setup,def:uc-emulation}
An execution setup with corruption pattern \(\Corr\) exists only if
\(\Corr\subseteq\Ids(\Prot)\) for its protocol.  Consequently, an emulation
statement at \(\Corr\) has nonvacuous real and ideal setup instances only when
\(\Corr\) is a subset of both protocols' identity sets.  The current
\(\mathsf{UCEmulatesAt}\) declaration does not state these two subset premises
at its outermost level; instead, it quantifies over setup records and uses
implications equating their stored patterns to \(\Corr\).
\end{definition}

\begin{definition}[UC realization]
\label{def:uc-realization}
\uses{def:uc-ideal-protocol,def:uc-emulation}
\lean{LeanCryptoProtocols.UC.UCRealizesAt, LeanCryptoProtocols.UC.UCRealizesPerfect, LeanCryptoProtocols.UC.UCRealizesPerfectWithCorruption, LeanCryptoProtocols.UC.UCRealizesStatistical, LeanCryptoProtocols.UC.UCRealizesStatisticalWithCorruption, LeanCryptoProtocols.UC.UCRealizesComputational, LeanCryptoProtocols.UC.UCRealizesComputationalWithCorruption}
\leanok
A protocol \(\Prot\) UC-realizes an ideal functionality \(F\) at level
\(\ell\) and corruption pattern \(\Corr\) exactly when it UC-emulates the
generated ideal protocol:
\[
  \UCReal_\ell(\Corr,\Prot,F)
  \Longleftrightarrow
  \UCEmul_\ell(\Corr,\Prot,\Phi_F).
\]
The ideal setup uses the same environment and the same fixed corruption pattern
as the real setup.  The simulator does not replace or directly program dummy
parties; all effective corrupted-party behavior in the ideal world is defined
by the program of \(F\).
\end{definition}

\begin{definition}[Computational-model boundary]
\label{rem:uc-computational-boundary}
\uses{def:uc-ppt,def:uc-controller-ppt,def:uc-emulation}
Computational security currently assumes the abstract \(\PPTM\) predicate and
its closure axioms.  It does not yet derive polynomial running time from the
machine programs, nor does it require protocol machines themselves to carry
PPT proofs.  Combined with the fixed controller budget, this is an explicit
abstraction boundary that later cryptographic reductions must respect.
\end{definition}

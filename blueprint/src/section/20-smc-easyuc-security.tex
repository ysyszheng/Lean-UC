This section instantiates the UC Core model with the current
\leanid{SMCEasyUC} case study.  It names the SMC payloads, local
functionalities, real certificate protocol, and proof obligations needed for
the UC security theorem.  Items stated as obligations are proof objects that
should be added to Lean; they are not claims that the repository already proves.

\subsection{Payloads and local functionality behavior}

\begin{definition}[Case-study payload]
The shared payload type is \(\Payload\), defined in the SMC common
functionality file.  It has three independent families:
\[
\begin{array}{ll}
\hbox{plain:}&
  \leanid{forw\_plain}\\
& \leanid{ke\_plain}\\
& \leanid{smc\_plain}\\[0.2em]
\hbox{to functionality:}&
  \leanid{forw\_to\_functionality}\\
& \leanid{ke\_to\_functionality}\\
& \leanid{smc\_to\_functionality}\\[0.2em]
\hbox{from functionality:}&
  \leanid{forw\_from\_functionality}\\
& \leanid{ke\_from\_functionality}\\
& \leanid{smc\_from\_functionality}.
\end{array}
\]
The business messages are:
\[
\begin{array}{ll}
\leanid{ForwBody}:&
  \leanid{submit},\ \leanid{observe},\ \leanid{release},\
  \leanid{delivered},\\[0.2em]
\leanid{KEBody}:&
  \leanid{init},\ \leanid{confirm},\ \leanid{key},\\
& \leanid{release\_init},\ \leanid{release\_confirm},\\
& \leanid{observe\_init\_share},\\
& \leanid{observe\_confirm\_share},\\[0.2em]
\leanid{SMCBody}:&
  \leanid{send},\ \leanid{observe},\ \leanid{release},\
  \leanid{received}.
\end{array}
\]
The proof must pattern match on these constructors.  There is no separate
address parser and no separate message-kind parser.
\end{definition}

\begin{definition}[Forwarding functionality]
\(\leanid{IdealForw}\) is the ideal forwarding functionality.  Its machine has
state
\[
  \leanid{fstate1},\qquad
  \leanid{fstate2}\;q,\qquad
  \leanid{fstate3},
\]
where \(q\) is a pending network message containing sender, receiver, and a
\(\leanid{NetworkBody}\).

In \(\leanid{fstate1}\), the accepted input shapes are exactly:
\[
\begin{array}{ll}
\leanid{forw\_plain}(\leanid{submit}\;s\;r\;x)
  &\hbox{with }m.\leanid{source}=\leanid{some}\;s,\\
\leanid{forw\_to\_functionality}\;c\;(\leanid{submit}\;s\;r\;x)
  &\hbox{with }c=\leanid{some}\;s .
\end{array}
\]
The next resume emits one backdoor envelope to \(\AdvId\) with payload
\[
  \leanid{forw\_plain}(\leanid{observe}\;s\;r\;x)
\]
and stores \(q=(s,r,x)\).  In \(\leanid{fstate2}\), an adversary message
\[
\begin{aligned}
  m.\leanid{source}&=\leanid{some}\;\AdvId,\\
  m.\leanid{label}&=\leanid{backdoor},\\
  m.\leanid{payload}&=\leanid{forw\_plain}\;\leanid{release}
\end{aligned}
\]
moves the machine to \(\leanid{fstate3}\); the next resume emits a
subroutine-output envelope to the receiver dummy with payload
\[
  \leanid{forw\_from\_functionality}\;r_{\mathsf{ext}}\;
    (\leanid{delivered}\;s\;r\;x).
\]
All other messages leave the state unchanged.
\end{definition}

\begin{definition}[Ideal secure-message functionality]
Write
\[
\begin{aligned}
  \IdealSMC &=
    \leanid{IdealSMC}\;\leanid{smc\_ids},\\
  \leanid{ideal\_smc\_functionality} &= \IdealSMC .
\end{aligned}
\]
It has parties
\[
  s_S=\leanid{smc\_sender\_id},\qquad
  r_S=\leanid{smc\_receiver\_id}
\]
and functionality identity \(f_I=\leanid{ideal\_smc\_id}\).  The functionality
state is
\[
  \leanid{sstate1},\qquad
  \leanid{sstate2}\;q,\qquad
  \leanid{sstate3},
\]
where \(q=(\sid,s_{\mathsf{ext}},r_{\mathsf{ext}},m)\) stores the plaintext.

In \(\leanid{sstate1}\), it accepts either the direct party message
\[
  \leanid{smc\_plain}(\leanid{send}\;\sid\;m)
\]
from \(s_S\), or the dummy-party wrapped message
\[
  \leanid{smc\_to\_functionality}
    \;(\leanid{some}\;s_{\mathsf{ext}})
    \;(\leanid{send}\;\sid\;m).
\]
Use the external-party abbreviations
\[
\begin{array}{rcl}
s_{\mathsf{ext}}&=&\leanid{sender\_external\_id},\\
r_{\mathsf{ext}}&=&\leanid{receiver\_external\_id}.
\end{array}
\]
The next resume reveals only
\[
  \leanid{smc\_plain}
    (\leanid{observe}\;\sid\;s_{\mathsf{ext}}\;r_{\mathsf{ext}})
\]
on the backdoor port.  The plaintext is not sent to the adversary.

In \(\leanid{sstate2}\;q\), a backdoor release from \(\AdvId\) with the same
session id moves to \(\leanid{sstate3}\).  The next resume emits
\[
  \leanid{smc\_from\_functionality}\;r_{\mathsf{ext}}\;
    (\leanid{received}\;\sid\;m)
\]
to the receiver dummy.  Wrong session ids and all non-matching payloads are
ignored.
\end{definition}

\begin{definition}[Toy encryption layer]
The current case study does not use the group-valued one-time pad from the old
blueprint.  It uses:
\[
\begin{aligned}
\leanid{key\_bit}(k) &=(k.\leanid{value}\bmod 2=1),\\
\leanid{enc}(k,m) &= m.\leanid{value}\oplus \leanid{key\_bit}(k),\\
\leanid{dec}(k,c) &= c.\leanid{value}\oplus \leanid{key\_bit}(k).
\end{aligned}
\]
The Lean theorem already available is
\[
  \leanid{dec\_enc}\;k\;m :
  \leanid{dec}\;k\;(\leanid{enc}\;k\;m)=m.
\]
Plaintext hiding is not a consequence of \(\leanid{dec\_enc}\).  The final SMC
proof needs an additional masking lemma for the distribution of
\(\leanid{enc}\;K\;m\), stated below as an obligation.
\end{definition}

\subsection{Real SMC protocol certificate}

\begin{definition}[Fixed certificate identities]
The certificate object file fixes the following identities:
\[
\begin{array}{c|l}
10 & \leanid{smc\_sender\_id}\\
11 & \leanid{smc\_receiver\_id}\\
12 & \leanid{ke\_sender\_id}\\
13 & \leanid{ke\_receiver\_id}\\
14 & \leanid{forw\_ke\_forward\_id}\\
15 & \leanid{forw\_ke\_return\_id}\\
16 & \leanid{forw\_smc\_id}\\
17 & \leanid{ideal\_smc\_id}\\
18 & \leanid{sender\_external\_id}\\
19 & \leanid{receiver\_external\_id}.
\end{array}
\]
The real protocol contains the first seven machines.  The core no longer has
a global visibility set.  In the SMC proof, the meaningful network-control
interfaces are the three forwarding functionalities
\[
  \{
    \leanid{forw\_ke\_forward\_id},
    \leanid{forw\_ke\_return\_id},
    \leanid{forw\_smc\_id}
  \}.
\]
We write these three forwarding identities as
\[
  F_{\to},\qquad F_{\leftarrow},\qquad F_S
\]
when the exact Lean names have already been displayed.
The controller's runtime overlay can route ordinary \(\leanid{plain}\)
backdoor traffic to every protocol machine; the effect of such traffic is still
determined by the target program.  The SMC argument only uses the forwarding
programs' observe/release behavior as adversarial network scheduling.
\end{definition}

\begin{definition}[Real protocol machines]
For a group generator \(g\), the real certificate protocol is
\[
  \RealSMC_g=\leanid{real\_smc\_protocol}\;g .
\]
It contains:
\[
\begin{array}{ll}
\hbox{main machines:}&
  \leanid{smc\_sender\_machine},\quad
  \leanid{smc\_receiver\_machine},\\[0.2em]
\hbox{internal machines:}&
  \leanid{ke\_sender\_machine}\;g,\quad
  \leanid{ke\_receiver\_machine}\;g,\\
&
  (\leanid{IdealForw}\;\leanid{forw\_ke\_forward\_ids}).\leanid{machine},\\
&
  (\leanid{IdealForw}\;\leanid{forw\_ke\_return\_ids}).\leanid{machine},\\
&
  (\leanid{IdealForw}\;\leanid{forw\_smc\_ids}).\leanid{machine}.
\end{array}
\]
The certificate proof file already proves unique ids, caller/subroutine
matching, separation from \(\EnvId,\AdvId\), backdoor-only adversary
communication, and that the two main machines are \(s_S\) and \(r_S\).
\end{definition}

\begin{definition}[Real sender and receiver programs]
The sender program has state
\[
  (\leanid{waiting\_message},\leanid{pending\_outgoing}).
\]
On \(\leanid{smc\_plain}(\leanid{send}\;\sid\;m)\) it stores
\((\sid,m)\) and sends \(\leanid{ke\_plain}\;\leanid{init}\) to the KE sender.
On \(\leanid{ke\_plain}(\leanid{key}\;k)\), if a message is waiting, it sends to
\(F_S\).  With
\[
  C_{\sid}=\leanid{smc\_cipher}\;\sid\;(\leanid{enc}\;k\;m),
\]
the submitted forwarding payload is
\[
  \leanid{forw\_plain}
    (\leanid{submit}\;s_S\;r_S\;C_{\sid}).
\]

The receiver program has state
\[
  (\leanid{session\_key},\leanid{pending\_outgoing}).
\]
On \(\leanid{ke\_plain}(\leanid{key}\;k)\) it stores \(k\).  On delivery of
\(\leanid{smc\_cipher}\;\sid\;c\), it uses
\[
  k=\leanid{session\_key.getD}\;\leanid{default\_shared\_key}
\]
and emits
\[
  \leanid{smc\_plain}(\leanid{received}\;\sid\;(\leanid{dec}\;k\;c))
\]
to \(r_{\mathsf{ext}}\).
\end{definition}

\begin{obligation}[Key-before-cipher invariant]
The full SMC proof must establish, for all legal executions used by the
security theorem, that every accepted SMC forwarding ciphertext delivery to the
receiver occurs only after the receiver has stored the session key output by the
KE receiver machine.

Equivalently, in every controller state reachable before the receiver handles
a delivered SMC ciphertext
\[
  \leanid{delivered}\;-\;-\;(\leanid{smc\_cipher}\;\sid\;c),
\]
the receiver local state must satisfy
\[
  \leanid{session\_key}=\leanid{some}\;k
\]
for the same challenge key \(k\) used by the sender.  If this invariant is not
provable with the current programs, the program must be changed so that the KE
receiver's pending key envelope is sent before the SMC ciphertext can be
released.
\end{obligation}

\subsection{Security statements}

\begin{definition}[Current certificate theorem statement]
The certificate target in the final certificate file is
\[
\begin{aligned}
&\leanid{ppt\_ddh\_assumption}\;gen \to {}\\
&\quad
  \UCComp(\RealSMC_{gen},\IdealSMC).
\end{aligned}
\]
Here \(\UCComp\) abbreviates the Lean predicate
\(\leanid{UCRealizesComputational}\), that is,
\[
  \leanid{UCRealizesAt}\;\leanid{computational}\;\varnothing\;
    \RealSMC_{gen}\;\IdealSMC .
\]
The fixed-simulator audit statement expands this to:
\[
\begin{aligned}
&\leanid{ppt\_ddh\_assumption}\;gen \to
  \forall A,E,\; \leanid{PPT}\;A \to \leanid{PPT}\;E \to {}\\
&\quad
  \forall \sigma_r,\sigma_i,\;
  \sigma_r.\leanid{corrupted\_parties}=\varnothing \to
  \sigma_i.\leanid{corrupted\_parties}=\varnothing \to {}\\
&\quad
  \exists \epsilon,\;
    \leanid{Negligible}\;\epsilon \ \land {}\\
&\quad
  \forall n,\;
    \ExecDiff(\sigma_r,\sigma_i,n)
    \le \epsilon(n),
\end{aligned}
\]
where
\[
\begin{aligned}
\sigma_r :
  \leanid{ExecutionSetup}\;
    \RealSMC_{gen}\;A\;E,\\[0.2em]
\sigma_i :
  \leanid{ExecutionSetup}\;
    \IdealProt_{\IdealSMC}\;
    (\leanid{smc\_simulator}\;A)\;E .
\end{aligned}
\]
\end{definition}

\begin{obligation}[Simulator is not the placeholder]
The current Lean definition
\[
  \leanid{smc\_simulator}\;A=A
\]
is only an interface placeholder.  It cannot be the final simulator for
\(\IdealSMC\), because the real execution's meaningful network-control
behavior is implemented by three forwarding functionalities, while the ideal
execution must reproduce the corresponding behavior through the ideal SMC
functionality's backdoor/control interface.  The generated dummy parties do not
give the simulator a separate control surface.

The final simulator must be a new adversary machine with identity \(\AdvId\).
Its local state must contain:
\begin{itemize}
\item the wrapped state of \(A\);
\item the stage of the simulated KE transcript;
\item the pending simulated observations for \(F_{\to}\), \(F_{\leftarrow}\),
and \(F_S\);
\item the SMC session id and external sender/receiver ids learned from
  \(\IdealSMC\)'s observe message;
\item the sampled or challenge-programmed ciphertext shown to \(A\).
\end{itemize}
It must translate releases from \(A\) as follows:
\[
\begin{array}{rcl}
\mathsf{release}(F_{\to})
  &\mapsto& \hbox{advance simulated KE first delivery},\\
\mathsf{release}(F_{\leftarrow})
  &\mapsto& \hbox{advance simulated KE second delivery},\\
\mathsf{release}(F_S)
  &\mapsto& \leanid{release}\;\sid
             \hbox{ sent to }f_I.
\end{array}
\]
\end{obligation}

\subsection{DDH challenge layer already present}

\begin{definition}[DHKE challenge programming]
The DHKE subdirectory defines DDH challenge values:
\[
\begin{array}{rcl}
\leanid{challenge\_first\_share}(X,Y,Z)&=&X,\\
\leanid{challenge\_second\_share}(X,Y,Z)&=&Y,\\
\leanid{challenge\_shared\_key}(X,Y,Z)&=&Z.
\end{array}
\]
It also defines challenge-programmed KE sender/receiver machines, challenge
real protocols, challenge ideal protocols, and setup transport lemmas.  The
available controller execution wrapper is
\[
  \leanid{ControllerExecution}
    =(\pi,A,E,\sigma).
\]
For a real DHKE setup \(\sigma\), the induced DDH distinguisher is
\[
  D^{\mathsf{dh}}_{\sigma}
  =\leanid{ddh\_adversary\_of\_real\_setup}\;g\;\sigma .
\]
\end{definition}

\begin{proposition}[Existing DDH milestone]
The concrete DHKE security theorem in the repository
proves the following.  If \(gen\) satisfies
\(\leanid{ppt\_ddh\_assumption}\), \(A\) and \(E\) are PPT, and
\[
  \sigma :
    \leanid{ExecutionSetup}\;
      (\leanid{DHKE.real\_protocol}\;gen.\leanid{run})\;A\;E,
\]
then there exists a negligible \(\epsilon\) such that
\[
  \forall n,\quad
  \leanid{challenge\_exec\_diff}\;gen.\leanid{run}\;\sigma\;n
    \le \epsilon(n).
\]
\end{proposition}

\begin{proof}
This is a direct application of the DDH assumption to the distinguisher
\(D^{\mathsf{dh}}_{\sigma}\).  The proof uses the controller PPT closure lemma
to show that the
challenge-programmed controller is PPT, and the displayed inequality is the
definition of the DDH distinguishing advantage after unfolding the two DDH
game equalities.
\end{proof}

\subsection{DHKE UC realization from DDH}

\begin{definition}[DHKE real and ideal objects]
For a group generator \(g\), write
\[
\begin{array}{rcl}
\RealKE_g
  &=&
  \leanid{DHKE.real\_protocol}\;g,\\[0.2em]
\IdealKE_g
  &=&
  \leanid{DHKE.ideal\_ke\_functionality}\;g,\\[0.2em]
\PhiKE_g
  &=&
  (\leanid{mk\_ideal\_protocol}\;\IdealKE_g).\leanid{protocol},\\[0.2em]
\SimKE_g(A)
  &=&
  \leanid{DHKE.simulator}\;g\;A .
\end{array}
\]
The target statement exposed by Lean is
\[
\begin{aligned}
  \leanid{DHKE.uc\_realizes}\;gen
  &=
  \leanid{ppt\_ddh\_assumption}\;gen\\
  &\quad\to
  \leanid{UCRealizesComputational}\;
    \RealKE_{gen.\leanid{run}}\;\IdealKE_{gen.\leanid{run}} .
\end{aligned}
\]
Unfolding the no-corruption alias, the theorem to prove is
\[
  \leanid{UCRealizesAt}\;
    \leanid{computational}\;\varnothing\;
    \RealKE_g\;\IdealKE_g .
\]
Thus, for every PPT adversary \(A\), the simulator must be
\(\SimKE_g(A)\); for every PPT environment \(E\) and every legal pair of
setups
\[
  \sigma_r:\leanid{ExecutionSetup}\;\RealKE_g\;A\;E,\qquad
  \sigma_i:\leanid{ExecutionSetup}\;\PhiKE_g\;(\SimKE_g(A))\;E,
\]
with
\[
  \sigma_r.\leanid{corrupted\_parties}
  =
  \sigma_i.\leanid{corrupted\_parties}
  =
  \varnothing ,
\]
one must exhibit a negligible function \(\epsilon\) such that
\[
  \forall n,\qquad
  \ExecDiff(\sigma_r,\sigma_i,n)\le \epsilon(n).
\]

The intended \(\IdealKE\) interface is not a DH-transcript functionality.
It does not sample or reveal the DH public shares.  Instead, it sends the
simulator two scheduling notifications: the first says that the first KE
direction is ready to be released, and the second says that the confirmation
direction is ready to be released.  Each notification contains only the
metadata needed to identify the session and the parties, not a group element.
The simulator decides whether to release the corresponding ideal key output.
The two public DH shares seen by the real-world adversary are simulated by
\(\SimKE_g(A)\), not leaked by \(\IdealKE_g\).
\end{definition}

\begin{definition}[DHKE hybrids]
Fix \(g,A,E,\sigma_r,\sigma_i\) as above.  We use the abbreviations
\[
\begin{array}{rcl}
\ChalROut(g,\sigma,x,n)
  &=&
  \leanid{challenge\_controller\_output\_of\_real\_setup}\;
    g\;\sigma\;x\;n,\\[0.2em]
\ChalIOut(g,\sigma,x,n)
  &=&
  \leanid{challenge\_controller\_output\_of\_ideal\_setup}\;
    g\;\sigma\;x\;n .
\end{array}
\]
Then set
\[
\begin{array}{rcl}
\Hyb_0(n)
  &=&
  \Ctrl.\leanid{exec}\;\sigma_r\;n,\\[0.2em]
\Hyb_1(n)
  &=&
  (\leanid{ddh\_real\_witness}\;g\;n).\leanid{bind}\;
    \lambda w,\;
      \ChalROut(g,\sigma_r,w^\circ,n),\\[0.2em]
\Hyb_2(n)
  &=&
  (\leanid{ddh\_real}\;g\;n).\leanid{bind}\;
    \lambda x,\;
      \ChalROut(g,\sigma_r,x,n),\\[0.2em]
\Hyb_3(n)
  &=&
  (\leanid{ddh\_random}\;g\;n).\leanid{bind}\;
    \lambda x,\;
      \ChalROut(g,\sigma_r,x,n),\\[0.2em]
\Hyb_4(n)
  &=&
  (\leanid{ddh\_random}\;g\;n).\leanid{bind}\;
    \lambda x,\;
      \ChalIOut(g,\sigma_i,x,n),\\[0.2em]
\Hyb_5(n)
  &=&
  \Ctrl.\leanid{exec}\;\sigma_i\;n .
\end{array}
\]
Here \(w^\circ=\leanid{DDHRealWitness.to\_sample}\;w\).  The only
computational step is the passage from \(\Hyb_2\) to \(\Hyb_3\).  All other
steps must be exact equalities of \(\mathsf{PMF}\;\mathsf{Bool}\).
In \(\Hyb_4\), the DDH-random sample \(x=(G,X,Y,Z)\) is used only as a proof
coupling source: \(X\) and \(Y\) program the simulator's fake forwarding
observations, while \(Z\) programs the ideal key sampler.  The ideal
functionality itself must not send \(X\) or \(Y\) to the simulator.
\end{definition}

\begin{lemma}[Real execution equals DDH-real challenge]
For every \(n\),
\[
  \Hyb_0(n)=\Hyb_1(n)=\Hyb_2(n).
\]
\end{lemma}

\begin{proof}
First unfold the real protocol initial-state sampler.  In Lean this is the
theorem
\[
  \leanid{controller\_exec\_real\_eq\_real\_witness\_controller\_output},
\]
which rewrites \(\Ctrl.\leanid{exec}\;\sigma_r\) as a bind over
\(\leanid{ddh\_real\_witness}\).  It remains to prove, for every witness \(w\),
that the real protocol initialized with \(w\)'s local DH secrets has the same
controller output as the challenge-real protocol initialized with
\[
  X=g^a,\qquad Y=g^b,\qquad Z=g^{ab}.
\]

This second equality is a fuel induction on
\(\leanid{LeanCryptoProtocols.max\_controller\_steps}\).  The state relation
relates a real controller state and a challenge-real controller state by:
\begin{itemize}
\item equal environment local state and equal adversary local state;
\item the same active machine identity;
\item syntactically equal states for the two forwarding machines;
\item related initiator states: the real secret is \(a\), the first outgoing
share is \(X\), and the final initiator key is \(Z\);
\item related responder states: the real secret is \(b\), the second outgoing
share is \(Y\), and the final responder key is \(Z\);
\item identical pending envelopes after rewriting by the local DH lemmas.
\end{itemize}
The local rewrites required in the induction are exactly:
\[
\begin{array}{c}
\leanid{challenge\_first\_share\_of\_witness},\\
\leanid{challenge\_second\_share\_of\_witness},\\
\leanid{initiator\_derive\_key\_eq\_challenge\_shared\_key},\\
\leanid{responder\_derive\_key\_eq\_challenge\_shared\_key},\\
\leanid{initiator\_resume\_send\_first\_of\_witness},\\
\leanid{responder\_resume\_send\_second\_of\_witness},\\
\leanid{initiator\_resume\_output\_key\_of\_witness}.
\end{array}
\]
Every controller-valid envelope on one side is therefore the same envelope on
the other side.  Invalid-envelope branches are also matched, because the
runtime ports, labels, owners, and source identities are syntactically the same
after the displayed rewrites.  The final environment output distribution is
therefore equal for each \(w\).

Finally, the witness distribution projects to the standard DDH-real
distribution:
\[
  (\leanid{ddh\_real\_witness}\;g\;n).\leanid{bind}
    (\lambda w,\mathsf{pure}\;w^\circ)
  =
  \leanid{ddh\_real}\;g\;n .
\]
This is the Lean theorem
\(\leanid{ddh\_real\_witness\_to\_sample\_eq\_ddh\_real}\), packaged at the
controller level as
\[
  \leanid{real\_witness\_challenge\_execution\_eq\_real\_challenge\_execution}.
\]
Thus \(\Hyb_1=\Hyb_2\).
\end{proof}

\begin{lemma}[DDH replacement]
Assume \(\leanid{ppt\_ddh\_assumption}\;gen\), \(g=gen.\leanid{run}\),
\(\leanid{PPT}\;A\), and \(\leanid{PPT}\;E\).  Then there is a negligible
\(\epsilon\) such that
\[
  \forall n,\qquad
  \left|
    \ProbOne(\Hyb_2(n))-\ProbOne(\Hyb_3(n))
  \right|
  \le \epsilon(n).
\]
\end{lemma}

\begin{proof}
Define the DDH distinguisher
\[
  D_{\sigma_r}(x,n)
  =
  \leanid{challenge\_controller\_output\_of\_real\_setup}\;
    g\;\sigma_r\;x\;n .
\]
The controller PPT closure theorem gives
\[
  \leanid{PPT}\;D_{\sigma_r},
\]
provided \(A\) and \(E\) are PPT.  This is the Lean theorem
\[
  \leanid{ppt\_ddh\_adversary\_of\_real\_setup}.
\]
By the DDH assumption, the outputs of \(D_{\sigma_r}\) on
\(\leanid{ddh\_real}\;g\) and \(\leanid{ddh\_random}\;g\) are
computationally indistinguishable, so there is negligible \(\epsilon\) with
\[
  \forall n,\quad
  \leanid{ddh\_advantage}\;g\;D_{\sigma_r}\;n\le \epsilon(n).
\]
Unfolding \(D_{\sigma_r}\), this is exactly the displayed bound between
\(\Hyb_2\) and \(\Hyb_3\).  In Lean, the packaged theorem is
\[
  \leanid{concrete\_ddh\_security\_negligible\_of\_real\_setup}.
\]
\end{proof}

\begin{lemma}[DDH-random challenge equals the ideal execution]
For every \(n\),
\[
  \Hyb_3(n)=\Hyb_4(n)=\Hyb_5(n).
\]
\end{lemma}

\begin{proof}
Fix a DDH-random sample \(x=(X,Y,Z)\).  The challenge-real protocol gives the
real adversary the following view:
\[
  \leanid{Forw.observe}\;X,\qquad
  \leanid{Forw.observe}\;Y,
\]
and outputs the same key \(Z\) to both KE parties, subject only to the
adversary's two release messages.

The challenge-ideal execution must reproduce this view without making
\(\IdealKE\) a transcript-aware functionality.  It is constructed from the
same sample \(x\) as follows.
\begin{itemize}
\item The simulator is challenge-programmed with \(X\) and \(Y\).  When
\(\IdealKE\) sends its first metadata-only scheduling notification, the
simulator delivers to the wrapped adversary \(A\) a forged
\(\leanid{Forw.observe}\;X\).  When \(\IdealKE\) sends its second scheduling
notification, the simulator delivers a forged \(\leanid{Forw.observe}\;Y\).
\item The ideal functionality is challenge-programmed only with the key
component \(Z\).  Its two notifications to the simulator reveal that a
corresponding KE direction is pending, but they do not reveal \(X\) or \(Y\).
\item The simulator stores \(A\)'s local state and translates the first and
second forwarding releases requested by \(A\) into
\[
  \leanid{IdealKE.release\_init},\qquad
  \leanid{IdealKE.release\_confirm}.
\]
These release commands make \(\IdealKE\) send key \(Z\) to the appropriate
dummy party, which then forwards the output to the environment.
\end{itemize}

The exact equality proof is again a bounded controller-state simulation.  The
state relation must record:
\begin{itemize}
\item equal environment state;
\item the real adversary state equals the simulator's wrapped adversary state;
\item the simulator stage is
\(\leanid{waiting\_init\_release}\),
\(\leanid{waiting\_confirm\_release}\), or \(\leanid{done}\), matching which
real forwarding release is next;
\item real forwarding states correspond to \(\IdealKE\)'s current release phase
and to the simulator's pending forged input to \(A\);
\item the simulator's forged first and second observations are exactly
\(X\) and \(Y\), respectively;
\item both parties' eventual output envelopes carry the same key
\(\leanid{challenge\_shared\_key}\;x=Z\);
\item every message delivered to \(E\) and every message delivered to the
wrapped \(A\) has the same payload, source convention, label, and destination
meaning on both sides.
\end{itemize}
The induction is allowed to use stuttering: one real step may correspond to a
bounded nonempty ideal path, and conversely.  The bound is a concrete constant
because the KE protocol and the two forwarding functionalities are one-shot.
No computational assumption is used in this lemma.

After the fixed-sample equality, \(\Hyb_4=\Hyb_5\) follows by unfolding the
standard ideal-world samplers.  In the corrected model this is not a sampler
for a full ideal KE transcript.  It is a joint-distribution statement saying
that independently running the standard simulator's fake-share sampler and the
standard \(\IdealKE\) key sampler has the same distribution as drawing
\(x\leftarrow\leanid{ddh\_random}\;g\;n\) and using \(X,Y\) for the simulator
and \(Z\) for the ideal key:
\[
\begin{aligned}
&\leanid{sample\_ideal\_challenge\_components}\;g\;n\\
&\quad =
  (\leanid{ddh\_random}\;g\;n).\leanid{bind}
    \lambda x,\;
      \mathsf{pure}\;C_x,\\
&\text{where}\quad
  C_x =
    (\leanid{challenge\_simulator\_view}\;x,
     \leanid{challenge\_ideal\_key}\;x).
\end{aligned}
\]
Lean must replace the old full-transcript sampler lemma with this
share/key-separated sampler lemma, and then lift it through
\(\leanid{mk\_ideal\_protocol}\), \(\leanid{ExecutionSetup}\), and
\(\Ctrl.\leanid{exec}\).
\end{proof}

\begin{theorem}[Conditional DHKE computational UC-realization of \(\IdealKE\)]
Let \(gen\) be a PPT group generator satisfying
\(\leanid{ppt\_ddh\_assumption}\;gen\).  If the exact trace obligations above
are proved for the corrected ideal interface, then
\[
  \leanid{UCRealizesComputational}\;
    \RealKE_{gen.\leanid{run}}\;\IdealKE_{gen.\leanid{run}} .
\]
\end{theorem}

\begin{proof}
Unfold \(\leanid{UCRealizesComputational}\).  Let \(A\) be any PPT adversary.
Choose
\[
  S=\SimKE_{gen.\leanid{run}}(A).
\]
The simulator is a machine with identity \(\AdvId\), with a single static
backdoor port to the environment; all communication with \(\IdealKE\) uses the
runtime backdoor overlay.  The proof must show \(\leanid{PPT}\;S\) from
\(\leanid{PPT}\;A\), because \(S\)'s program performs only finite state
bookkeeping around black-box calls to \(A\)'s receive and resume programs.

Now fix any PPT environment \(E\), legal setups \(\sigma_r,\sigma_i\), and the
no-corruption premises.  By the two exact trace lemmas and the DDH replacement
lemma, there is negligible \(\epsilon\) such that, for every \(n\),
\[
\begin{aligned}
\ExecDiff(\sigma_r,\sigma_i,n)
&=
\left|
  \ProbOne(\Hyb_0(n))-\ProbOne(\Hyb_5(n))
\right|\\
&=
\left|
  \ProbOne(\Hyb_2(n))-\ProbOne(\Hyb_3(n))
\right|\\
&\le \epsilon(n).
\end{aligned}
\]
This is precisely the computational branch of
\(\leanid{UCRealizesAt}\;\leanid{computational}\;\varnothing\).
\end{proof}

Current Lean already implements:

\(\leanid{DHKE.uc\_realizes}\), but only as a proposition-valued definition;
\item \(\RealKE_g\), \(\IdealKE_g\), and the black-box simulator
\(\SimKE_g(A)\);
\item DDH-real witnesses and their projection to \(\leanid{ddh\_real}\);
\item the challenge-real protocol and setup transport lemmas preserving
\(\leanid{corrupted\_parties}\);

\item the setup-specific DDH reduction theorem
\[
\begin{gathered}
\leanid{concrete\_ddh\_security}\\
\leanid{\_negligible\_of\_real\_setup}.
\end{gathered}
\]
\end{itemize}
Lean still needs the following theorem-level objects:
\begin{itemize}
\item \(\leanid{PPT}\;(\SimKE_g(A))\) from \(\leanid{PPT}\;A\);
\item a fixed-witness controller simulation proving
\(\Hyb_0=\Hyb_1\);
\item a fixed-DDH-random-sample controller simulation proving

\item a share/key-separated ideal-sampler lifting theorem proving

The most delicate missing piece is \(\Hyb_3=\Hyb_4\).  The implementation must

messages and must be forged by the simulator.  After this correction, the proof
must check that the two metadata-only \(\IdealKE\) notifications align with the
two real-world forwarding directions, and that the real DHKE responder's
scheduling of the second share and key output admits the required stuttering
simulation.
\end{obligation}

\subsection{Exact trace equalities required for SMC}

\begin{obligation}[Real SMC equals the DDH-real challenge]
For every legal real SMC setup \(\sigma_r\), prove equality of output
ensembles:
\[
\begin{aligned}
  \Ctrl.\leanid{exec}\;\sigma_r
  &=
  \lambda n,\;
    (\DDHWit(g,n)).\leanid{bind}\\
  &\quad
      (\lambda w,\;
        \SMCChalOut(g,\sigma_r,w^\circ,n)).
\end{aligned}
\]
Use the abbreviations
\[
\begin{array}{rcl}
w^\circ &=& w.\leanid{to\_sample},\\
\DDHWit(g,n)&=&\leanid{ddh\_real\_witness}\;g\;n .
\end{array}
\]
The proof must be by controller-state simulation.  At a fixed witness \(w\),
relate the real state and the challenge state by:
\begin{itemize}
\item identical environment state;
\item identical adversary state;
\item identical states of the two SMC machines and the three forwarding
machines;
\item the real KE secrets from \(w\) correspond to the key \(K(w^\circ)\),
where \(K\) is the Lean projection \(\leanid{challenge\_shared\_key}\);
\item the active machine id is the same;
\item pending envelopes are syntactically equal after replacing real DH values
by the challenge projections already proved in the DHKE lemmas.
\end{itemize}
For each controller step, the Lean proof should use the existing local DHKE
lemmas such as
\[
\begin{array}{c}
\leanid{initiator\_resume\_send\_first\_of\_witness}\\
\leanid{responder\_resume\_send\_second\_of\_witness}\\
\leanid{initiator\_resume\_output\_key\_of\_witness}
\end{array}
\]
Non-KE machines are
handled by reflexive unfolding of their receive/resume definitions.  Invalid
outgoing envelopes must be matched by the same
\[
  \leanid{handle\_outgoing\_some\_invalid}
\]
branch on both sides.
\end{obligation}

\begin{obligation}[DDH-real witness projection]
After proving the previous equality, project the witness distribution to the
standard DDH-real distribution:
\[
\begin{aligned}
&(\DDHWit(g,n)).\leanid{bind}\;
  \lambda w,\;
    \SMCChalOut(g,\sigma_r,w^\circ,n)\\
&\qquad =
(\DDHReal(g,n)).\leanid{bind}\;
  \lambda x,\;
    \SMCChalOut(g,\sigma_r,x,n).
\end{aligned}
\]
This is the SMC-context version of the existing DHKE witness-projection lemma.
\end{obligation}

\begin{obligation}[Random challenge equals ideal]
Let
\[
  \sigma_i :
    \leanid{ExecutionSetup}\;
      (\IdealProt_{\IdealSMC})\;(\SimSMC A)\;E
\]
be a legal ideal setup with
\(\sigma_i.\leanid{corrupted\_parties}=\varnothing\).
Prove:
\[
\begin{aligned}
&(\DDHRand(g,n)).\leanid{bind}\;
  \lambda x,\;
    \SMCChalOut(g,\sigma_r,x,n)\\
&\qquad =
  \Ctrl.\leanid{exec}\;\sigma_i\;n .
\end{aligned}
\]
This is the main path-matching theorem.  The relation between real-random and
ideal states must include:
\begin{itemize}
\item equal environment state;
\item the real adversary state equals the wrapped adversary state inside
\(\SimSMC A\);
\item the ideal SMC functionality phase stores exactly the plaintext and
session id that the real SMC sender stored before encryption;
\item simulated KE and forwarding stages inside \(\SimSMC A\) correspond to
the real challenge KE and \(\leanid{Forw}\) phases;
\item the receiver output pending in either execution is the same payload
  \(O_{\sid,m}\), where
\[
  O_{\sid,m}=
    \leanid{smc\_plain}(\leanid{received}\;\sid\;m).
\]
\end{itemize}
When the real-random execution emits a ciphertext, the ideal simulator emits a
sampled ciphertext with the same distribution.  When the real receiver later
decrypts, the proof uses \(\leanid{dec\_enc}\) and the receiver-key invariant to
show that the delivered plaintext is exactly \(m\).
\end{obligation}

\begin{obligation}[Ciphertext masking]
The previous obligation requires the following distributional lemma.  Let
\[
  K(x)=\leanid{challenge\_shared\_key}\;x
\]
for \(x\leftarrow \DDHRand(g,n)\).  There must be a sampler, written
\(\SimC(g,n)\), independent of the plaintext such that, for all plaintexts
\(m\),
\[
  \mathsf{PMF.map}\;(\lambda x,\leanid{enc}\;(K(x))\;m)\;(\DDHRand(g,n))
  =
  \SimC(g,n).
\]
A Lean implementation can use the name
\(\leanid{sample\_simulated\_ciphertext}\).
If this cannot be proved from \(\leanid{GroupDescription}\), the model must add
an explicit encryption-hiding assumption or replace the toy encryption with a
one-time-pad interface whose key distribution is uniform in Lean.
\end{obligation}

\begin{obligation}[Release order and multi-step matching]
The random-to-ideal equality must not rely on informal scheduling.  Define a
finite matching relation for controller states:
\[
  R(s_r,s_i)
\]
and prove, for every related pair, one of the following precise step clauses:
\[
\begin{aligned}
  S_r(s_r)&=\Ctrl.\leanid{step}\;\sigma_r\;s_r,\\
  S_i(s_i)&=\Ctrl.\leanid{step}\;\sigma_i\;s_i .
\end{aligned}
\]
The lockstep case proves
\[
  S_r(s_r)=\mathsf{PMF.map}\;R^{-1}(S_i(s_i)).
\]
The real-stutter case matches one real step by a bounded nonempty ideal path.
The ideal-stutter case matches one ideal step by a bounded nonempty real path.
The bound must be a concrete natural number derived from the number of pending
simulated messages.  This allows the final Lean theorem to be an induction on
the controller fuel constant, not on an informal trace.  The relevant Lean
constant is \(\leanid{LeanCryptoProtocols.max\_controller\_steps}\).
\end{obligation}

\subsection{Final computational theorem}

\begin{theorem}[SMC UC theorem]
Assume the obligations above have been implemented.  For every PPT group
generator \(gen\), if it satisfies the Lean DDH assumption
\(\leanid{ppt\_ddh\_assumption}\;gen\), then
\[
  \UCComp(\RealSMC_{gen},\IdealSMC).
\]
\end{theorem}

\begin{proof}
Fix a PPT adversary \(A\) and a PPT environment \(E\).  Let
\[
  \sigma_r :
    \leanid{ExecutionSetup}\;
      \RealSMC_{gen}\;A\;E
\]
be any legal real setup.  Use the implemented SMC simulator to obtain
\(\SimSMC A\), and let \(\sigma_i\) be any legal ideal setup for
\(\IdealProt_{\IdealSMC}\), \(\SimSMC A\), and \(E\).  In the
\(\leanid{UCRealizesComputational}\) alias, both setups satisfy
\[
  \sigma_r.\leanid{corrupted\_parties}
  =
  \sigma_i.\leanid{corrupted\_parties}
  =
  \varnothing .
\]

By the real-trace equality and witness projection obligations,
\[
  \Ctrl.\leanid{exec}\;\sigma_r
  =
  \ChalReal(g,\sigma_r),
\]
where
\[
\begin{aligned}
  \ChalReal(g,\sigma_r)(n)
  &=
  (\DDHReal(g,n)).\leanid{bind}\;
    (\lambda x,\;\SMCChalOut(g,\sigma_r,x,n)).
\end{aligned}
\]
By the random-to-ideal path-matching obligation,
\[
  \ChalRand(g,\sigma_r)=\Ctrl.\leanid{exec}\;\sigma_i .
\]
Therefore
\[
\begin{aligned}
&\ExecDiff(\sigma_r,\sigma_i,n)\\
&\quad =
\left|
  \ProbOne(\ChalReal(g,\sigma_r)(n))
  -
  \ProbOne(\ChalRand(g,\sigma_r)(n))
\right|.
\end{aligned}
\]
Define the DDH distinguisher
\[
  D_{\sigma_r}(x,n)
  =
  \SMCChalOut(g,\sigma_r,x,n).
\]
The SMC-context PPT closure obligation gives \(\leanid{PPT}\;D_{\sigma_r}\).
Apply the DDH assumption for \(gen\) to \(D_{\sigma_r}\).  This yields a
negligible \(\epsilon\) such that, for every \(n\), the displayed difference is
at most \(\epsilon(n)\).  This is exactly the computational branch of the UC
realization predicate for the ideal SMC functionality.
\end{proof}
